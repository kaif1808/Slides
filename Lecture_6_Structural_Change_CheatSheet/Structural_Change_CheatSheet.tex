\subsection*{Engel's Law}
As income increases, the percentage spent on food (necessity) decreases, even though total food expenditure increases.

\section{Stone-Geary Preferences}
%==============================================================================

\subsection*{Utility Function}
\[
c_t = (c_t^A - \gamma^A)^{\eta_A} c_t^{M^{\eta_M}} (c_t^S + \gamma^S)^{\eta_S}
\tag{Stone-Geary}
\]

\medskip
\noindent where $\gamma^A, \gamma^S > 0$ are subsistence levels and $\eta_A + \eta_M + \eta_S = 1$.

\medskip
\noindent \textbf{Special cases:}
\begin{itemize}
    \item $\gamma_i = 0, \forall i$: Cobb-Douglas (homothetic)
    \item $\gamma^A > 0$: food is a necessity
    \item $\gamma^S > 0$: services is a luxury (negative threshold)
\end{itemize}

\subsection*{Demand Derivation}
\begin{derivationbox}{Derivation of Stone-Geary Demand}
Household solves:
\[
\max_{\{c_i\}} \mathcal{L} = \prod_i (c_i - \gamma_i)^{\eta_i} + \lambda\left[I - \sum_i p_i c_i\right]
\]

FOC:
\[
\eta_i (c_i - \gamma_i)^{-1} \prod_k (c_k - \gamma_k)^{\eta_k} = \lambda p_i
\]

Dividing FOC for $i$ by FOC for $j$:
\[
\frac{\eta_i}{\eta_j} \frac{c_j - \gamma_j}{c_i - \gamma_i} = \frac{p_i}{p_j}
\]

Solving the system:
\[
c_j = \gamma_j + \frac{\eta_j (I - \sum_i p_i \gamma_i)}{p_j}
\tag{Stone-Geary Demand}
\]
\end{derivationbox}

\begin{resultbox}
\[
\boxed{c_j = \gamma_j + \frac{\eta_j (I - \sum_i p_i \gamma_i)}{p_j}}
\tag{Demand Schedule}
\]
\end{resultbox}

\medskip
\noindent \textbf{Key insight:} Engel curve is linear but shifted by subsistence level. When income is low, only necessity goods are consumed.

\clearpage

\section{Demand-Side Model: Production and Equilibrium}
%==============================================================================

\subsection*{Sectoral Production Functions}
\[
Y_{it} = B_i F(K_{it}, X_t L_{it}) \quad i = \{A, M, S\}
\tag{Sectoral Production}
\]

\medskip
\noindent where $B_i$ is sector-specific TFP and $X_t$ grows at rate $g$.

\subsection*{Market Clearing}
\[
\sum_i K_{it} = K_t, \quad \sum_i L_{it} = L_t
\tag{Factor Market Clearing}
\]

\medskip
\begin{tcolorbox}[colback=yellow!10,colframe=yellow!60,title=Key Assumption]
Investment good is made from manufacturing good (no depreciation):
\[
Y_{Mt} = \dot{K}_t + c_t^M L_t
\]
\end{tcolorbox}

\subsection*{Price Relations (Manufacturing = Numeraire)}
\[
p_t^A = \frac{B_M}{B_A}, \quad p_t^M = 1, \quad p_t^S = \frac{B_M}{B_S}
\tag{Relative Prices}
\]

\subsection*{Competitive Factor Markets}
\[
w_t = MPL_{it}, \quad r_t = MPK_{it} \quad \forall i
\tag{Factor Price Equalization}
\]

This implies:
\[
\frac{K_{it}}{X_t L_{it}} = k_t \quad \forall i
\tag{Common Capital-Output Ratio}
\]

\clearpage

%==============================================================================
\section{Constant Growth Path (CGP)}
%==============================================================================

\subsection*{Definition}
A \textbf{Constant Growth Path (CGP)} is when the consumption aggregate grows at a constant rate but sectoral consumptions do not.

\medskip
\noindent \textbf{Existence condition:} CGP exists (unique) iff $\gamma^A / B_A = \gamma^S / B_S$.

\subsection*{Consumption Growth Rates on CGP}
\begin{derivationbox}{Derivation of Sectoral Growth Rates}
From intratemporal optimization:
\[
\eta^M \frac{B_M (c_t^A - \gamma^A)}{B_A \eta^A} = c_t^M = \eta^M \frac{B_M (c_t^S + \gamma^S)}{B_S \eta^S}
\]

With technology growth $g$ (i.e., $X_t = X_0 e^{gt}$):
\[
\frac{\dot{c}_t^A}{c_t^A} = g \frac{c_t^A - \gamma^A}{c_t^A} < g
\tag{Agriculture}
\]
\[
\frac{\dot{c}_t^M}{c_t^M} = g
\tag{Manufacturing}
\]
\[
\frac{\dot{c}_t^S}{c_t^S} = g \frac{c_t^S + \gamma^S}{c_t^S} > g
\tag{Services}
\]
\end{derivationbox}

\begin{resultbox}
\[
\boxed{\frac{\dot{c}^A}{c} < g < \frac{\dot{c}^S}{c} = g = \frac{\dot{c}^M}{c}}
\tag{CGP Consumption Growth}
\]
\end{resultbox}

\begin{intuitionbox}
As economy grows: agricultural consumption shrinks as share, manufacturing stays constant, services expand. This is Engel's Law in action.
\end{intuitionbox}

\clearpage

%==============================================================================
\section{Employment Dynamics (Demand Side)}
%==============================================================================

\subsection*{Employment Growth Rates}
\begin{derivationbox}{Derivation of Employment Reallocation}
Using factor market clearing and production functions:
\[
\frac{\dot{L}_{At}}{L_{At}} = n - g \frac{\gamma^A L_t}{B_A L_{At} X_t F(k^*,1)} < n
\tag{Agriculture}
\]
\[
\frac{\dot{L}_{Mt}}{L_{Mt}} = n
\tag{Manufacturing}
\]
\[
\frac{\dot{L}_{St}}{L_{St}} = n + g \frac{\gamma^S L_t}{B_S L_{St} X_t F(k^*,1)} > n
\tag{Services}
\]
\end{derivationbox}

\begin{resultbox}
\[
\boxed{\dot{L}_A/L_A < n < \dot{L}_S/L_S = n = \dot{L}_M/L_M}
\tag{Employment Growth}
\]
\end{resultbox}

\medskip
\begin{tcolorbox}[colback=blue!5,colframe=blue3,title=Key Result: Demand-Driven SC]
With Stone-Geary preferences (Engel's Law):
\begin{itemize}
    \item Agricultural employment share declines
    \item Service employment share rises
    \item Manufacturing employment share stays constant (at CGP)
\end{itemize}
\end{tcolorbox}

\clearpage

\section{Supply-Side Explanations: Baumol's Cost Disease}
%==============================================================================

\subsection*{Core Idea}
If two sectors grow at different rates and share common production factors, there will be reallocation between them. There cannot be a BGP with constant sectoral shares.

\medskip
\begin{intuitionbox}
Baumol's Cost Disease: Sectors with slower productivity growth see rising relative prices and absorb increasing labor shares because wages equalize across sectors.
\end{intuitionbox}

\subsection*{Productivity Growth Differentials (Duarte \& Restuccia)}
\begin{center}
\begin{tabular}{lc}
\toprule
Sector & Avg Growth Rate (1956-2004) \\
\midrule
Agriculture & 3.8\% \\
Manufacturing & 2.4\% \\
Services & 1.3\% \\
\bottomrule
\end{tabular}
\end{center}

\section{Supply-Side Model: CES Aggregator}
%==============================================================================

\subsection*{Final Good Aggregator (CES)}
\[
Y_t = \left[\omega Y_{1t}^{\frac{\epsilon-1}{\epsilon}} + (1-\omega) Y_{2t}^{\frac{\epsilon-1}{\epsilon}}\right]^{\frac{\epsilon}{\epsilon-1}}
\tag{CES Aggregator}
\]

\medskip
\noindent where $\epsilon > 0$ is the substitution elasticity.

\subsection*{Sectoral Production}
\[
Y_{it} = A_{it} K_{it}^{\alpha_i} L_{it}^{1-\alpha_i} \quad i = 1, 2
\tag{Cobb-Douglas Sectors}
\]

\medskip
\noindent Assume $\alpha_1 < \alpha_2$: sector 1 is more labor-intensive.

\subsection*{Productivity Growth}
\[
\frac{\dot{A}_{it}}{A_{it}} = a_i > 0
\tag{Exogenous Tech Growth}
\]

\clearpage

%==============================================================================
\section{Supply-Side: Factor Allocation}
%==============================================================================

\subsection*{Capital and Labor Shares}
Define:
\[
k_t \equiv \frac{K_{1t}}{K_t}, \quad \lambda_t \equiv \frac{L_{1t}}{L_t}
\tag{Allocation Fractions}
\]

\begin{derivationbox}{Derivation of Capital Allocation}
From equalized MPKs:
\[
\alpha_1 \omega Y_{1t}^{-\frac{1}{\epsilon}} \frac{Y_{1t}}{K_{1t}} = \alpha_2 (1-\omega) Y_{2t}^{-\frac{1}{\epsilon}} \frac{Y_{2t}}{K_{2t}}
\]

Simplifying:
\[
k_t = \left[1 + \frac{1-\omega}{\omega} \frac{\alpha_2}{\alpha_1} \left(\frac{Y_{2t}}{Y_{1t}}\right)^{\frac{\epsilon-1}{\epsilon}}\right]^{-1}
\tag{Capital Share}
\]

Similarly for labor:
\[
\lambda_t = \left[1 + \frac{1-\alpha_2}{1-\alpha_1} \frac{\alpha_1}{\alpha_2} \frac{1-k_t}{k_t}\right]^{-1}
\tag{Labor Share}
\]
\end{derivationbox}

\begin{resultbox}
\[
\boxed{k_t = \left[1 + \frac{1-\omega}{\omega} \frac{\alpha_2}{\alpha_1} \left(\frac{Y_{2t}}{Y_{1t}}\right)^{\frac{\epsilon-1}{\epsilon}}\right]^{-1}}
\tag{Capital Allocation}
\]
\end{resultbox}

\medskip
\noindent \textbf{Key property:} $\lambda_t$ is increasing in $k_t$; labor and capital flow to the same sector.

\clearpage

%==============================================================================
\section{Supply-Side: Response to Capital Accumulation}
%==============================================================================

\subsection*{Capital Response to Economy Expansion}
\begin{derivationbox}{Derivation of Capital Reallocation Response}
Taking logs and differentiating $k_t$ with respect to $K_t$:
\[
\frac{\partial \log k_t}{\partial \log K_t} > 0 \iff (\alpha_2 - \alpha_1)(1 - \epsilon) > 0
\tag{Capital Response}
\]

This implies:
\begin{itemize}
    \item If $\epsilon < 1$ (complements) and $\alpha_2 > \alpha_1$: capital flows to labor-intensive sector
    \item If $\epsilon > 1$ (substitutes) and $\alpha_2 > \alpha_1$: capital flows to capital-intensive sector
\end{itemize}
\end{derivationbox}

\begin{resultbox}
\[
\boxed{\frac{\partial \log k_t}{\partial \log K_t} > 0 \iff (\alpha_2 - \alpha_1)(1 - \epsilon) > 0}
\tag{Key Condition}
\]
\end{resultbox}

\subsection*{Response to Technology Changes}
\[
\frac{\partial \log k_t}{\partial \log A_{2t}} = -\frac{\partial \log k_t}{\partial \log A_{1t}} > 0 \iff \epsilon < 1
\tag{Tech Response}
\]

\medskip
\begin{intuitionbox}
When goods are complements ($\epsilon < 1$), faster technology growth in one sector causes capital to reallocate AWAY from that sector.
\end{intuitionbox}

\clearpage

%==============================================================================
\section{Supply-Side: Long-Run Dynamics}
%==============================================================================

\subsection*{Growth and Reallocation}
\begin{derivationbox}{Derivation of Sectoral Growth Relation}
From MRP equalization in logs, taking time derivatives:
\[
\frac{\epsilon-1}{\epsilon} \gamma_{Y_1} - \gamma_{K_1} = \frac{\epsilon-1}{\epsilon} \gamma_{Y_2} - \gamma_{K_2}
\]

This implies:
\[
\gamma_{K_1} - \gamma_{K_2} = \frac{\epsilon-1}{\epsilon} (\gamma_{Y_1} - \gamma_{Y_2})
\]

Therefore:
\[
\text{sign}(\gamma_{K_1} - \gamma_{K_2}) = \text{sign}(\gamma_{Y_1} - \gamma_{Y_2}) \iff \epsilon > 1
\]
\end{derivationbox}

\begin{resultbox}
\[
\boxed{
\begin{aligned}
&\gamma_{L_1} > \gamma_{L_2} \iff \gamma_{K_1} > \gamma_{K_2} \iff \gamma_{Y_1} < \gamma_{Y_2} \quad \text{if } \epsilon < 1 \\
&\gamma_{L_1} < \gamma_{L_2} \iff \gamma_{K_1} < \gamma_{K_2} \iff \gamma_{Y_1} < \gamma_{Y_2} \quad \text{if } \epsilon > 1
\end{aligned}}
\tag{Sectoral Growth Relation}
\]
\end{resultbox}

\section{Long-Run Equilibrium}
%==============================================================================

\subsection*{Factor Allocation in the Limit}
\begin{center}
\begin{tabular}{lp{0.4\linewidth}p{0.4\linewidth}}
\toprule
Condition & Goods Complements ($\epsilon < 1$) & Goods Substitutes ($\epsilon > 1$) \\
\midrule
Outcome & All factors go to slow-growing sector & All factors go to fast-growing sector \\
Growth rate & $\gamma = \min\{\gamma_{Y_1}, \gamma_{Y_2}\}$ & $\gamma = \max\{\gamma_{Y_1}, \gamma_{Y_2}\}$ \\
Mechanism & Rising relative price attracts labor & Productivity advantage dominates \\
\bottomrule
\end{tabular}
\end{center}

\subsection*{Role of Investment Sector}
If capital is made from one sector (e.g., manufacturing), in the limit:
\begin{itemize}
    \item Employment goes to slowest-growing non-capital-goods sector
    \item Capital-goods sector survives due to its role in capital creation
\end{itemize}

\section{Testing Demand vs Supply Explanations}
%==============================================================================

\subsection*{Identification Problem}
To distinguish demand-driven from supply-driven structural change, we need identifying moments:

\begin{center}
\begin{tabular}{lp{0.3\linewidth}p{0.3\linewidth}}
\toprule
Driver & Effect on Quantity & Effect on Price \\
\midrule
Demand increase & $\uparrow$ & $\uparrow$ \\
Productivity increase & $\uparrow$ & $\downarrow$ \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\begin{intuitionbox}
With both quantity and price data, we can distinguish demand from supply shocks. For structural change: changes in employment associated with income changes (demand) vs relative price changes (supply).
\end{intuitionbox}

\section{Problem-Solving Templates}
%==============================================================================

\subsection*{(a) Derive Demand from Stone-Geary Preferences}

\begin{enumerate}
    \item Set up Lagrangian: $\mathcal{L} = \prod_i (c_i - \gamma_i)^{\eta_i} + \lambda(I - \sum_i p_i c_i)$
    \item Take FOCs: $\eta_i (c_i - \gamma_i)^{-1} \prod_k (c_k - \gamma_k)^{\eta_k} = \lambda p_i$
    \item Divide FOC $i$ by FOC $j$ to get ratio form
    \item Solve system for $c_i$ in terms of $I$ and $p_i$
    \item Result: $c_i = \gamma_i + \frac{\eta_i (I - \sum_j p_j \gamma_j)}{p_i}$
\end{enumerate}

\subsection*{(b) Analyze CGP in Demand-Side Model}

\begin{enumerate}
    \item Check CGP existence: $\gamma^A/B_A = \gamma^S/B_S$
    \item From intratemporal FOCs, derive consumption ratios
    \item Use technology growth rate $g$ to find consumption growth rates
    \item Result: $\dot{c}^A/c < g < \dot{c}^S/c$
    \item Derive employment growth: $\dot{L}_A/L < n < \dot{L}_S/L$
\end{enumerate}

\subsection*{(c) Analyze Supply-Side Reallocation}

\begin{enumerate}
    \item Set up CES aggregator and sectoral production functions
    \item Derive factor allocation: $k_t$ and $\lambda_t$
    \item Compute responses to $K$ and $A$ changes
    \item Apply condition: $(\alpha_2 - \alpha_1)(1 - \epsilon) > 0$
    \item Determine long-run: complements $\Rightarrow$ slow sector wins
\end{enumerate}

\subsection*{(d) Compare Demand vs Supply Explanations}

\begin{enumerate}
    \item Identify key predictions of each theory
    \item Demand: Engel curves $\Rightarrow$ consumption shares shift with income
    \item Supply: productivity differentials $\Rightarrow$ relative price changes
    \item Find identifying moment: prices vs quantities
    \item Use data to assess relative importance
\end{enumerate}

\section{Common Pitfalls}
%==============================================================================

\begin{enumerate}
    \item \textbf{Confusing homothetic with non-homothetic}
    \begin{itemize}
        \item Homothetic: linear Engel curves, no demand-driven SC
        \item Non-homothetic: non-linear Engel curves, demand-driven SC possible
    \end{itemize}

    \item \textbf{Forgetting CGP existence condition}
    \begin{itemize}
        \item CGP exists iff $\gamma^A/B_A = \gamma^S/B_S$
        \item Without this, sectoral growth rates are not constant
    \end{itemize}

    \item \textbf{Misinterpreting complementarity}
    \begin{itemize}
        \item $\epsilon < 1$: goods are complements; faster tech $\Rightarrow$ reallocation AWAY
        \item $\epsilon > 1$: goods are substitutes; faster tech $\Rightarrow$ reallocation TOWARD
    \end{itemize}

    \item \textbf{Confusing sectoral growth with reallocation}
    \begin{itemize}
        \item Fast-growing sector can still see employment decline if reallocation is strong enough
    \end{itemize}

    \item \textbf{Ignoring the role of the investment sector}
    \begin{itemize}
        \item Sector that produces capital goods survives even if slow-growing
    \end{itemize}

    \item \textbf{Misinterpreting Engel curves}
    \begin{itemize}
        \item Necessity: share declines but absolute consumption rises
        \item Luxury: share rises and absolute consumption rises faster
    \end{itemize}

    \item \textbf{Confusing final expenditure with value-added}
    \begin{itemize}
        \item These give very different parameter estimates
        \item Affects conclusions about demand vs supply importance
    \end{itemize}
\end{enumerate}

\section{Quick Reference Formulas}
%==============================================================================

\subsection*{Stone-Geary Preferences}
\begin{align*}
c_t &= (c_t^A - \gamma^A)^{\eta_A} c_t^{M^{\eta_M}} (c_t^S + \gamma^S)^{\eta_S} \\
c_j &= \gamma_j + \frac{\eta_j (I - \sum_i p_i \gamma_i)}{p_j}
\end{align*}

\subsection*{Production}
\begin{align*}
Y_{it} &= B_i F(K_{it}, X_t L_{it}) \\
Y_t &= [\omega Y_{1t}^{\frac{\epsilon-1}{\epsilon}} + (1-\omega) Y_{2t}^{\frac{\epsilon-1}{\epsilon}}]^{\frac{\epsilon}{\epsilon-1}}
\end{align*}

\subsection*{CGP Growth Rates (Demand Side)}
\begin{align*}
\frac{\dot{c}^A}{c} &= g \frac{c^A - \gamma^A}{c^A} < g \\
\frac{\dot{c}^M}{c} &= g \\
\frac{\dot{c}^S}{c} &= g \frac{c^S + \gamma^S}{c^S} > g
\end{align*}

\subsection*{Employment Growth (Demand Side)}
\begin{align*}
\frac{\dot{L}_A}{L} &< n \\
\frac{\dot{L}_M}{L} &= n \\
\frac{\dot{L}_S}{L} &> n
\end{align*}

\subsection*{Capital Allocation (Supply Side)}
\begin{align*}
k_t &= \left[1 + \frac{1-\omega}{\omega} \frac{\alpha_2}{\alpha_1} \left(\frac{Y_{2t}}{Y_{1t}}\right)^{\frac{\epsilon-1}{\epsilon}}\right]^{-1} \\
\frac{\partial \log k}{\partial \log K} &> 0 \iff (\alpha_2 - \alpha_1)(1 - \epsilon) > 0
\end{align*}

\subsection*{Key Conditions}
\begin{itemize}
    \item CGP exists: $\gamma^A/B_A = \gamma^S/B_S$
    \item Capital response: $(\alpha_2 - \alpha_1)(1 - \epsilon) > 0$
    \item Long-run: $\epsilon < 1 \Rightarrow$ slow sector wins; $\epsilon > 1 \Rightarrow$ fast sector wins
\end{itemize}
