\section{Common Pitfalls and Tips}
%==============================================================================

\subsection*{Common Pitfalls}

\begin{enumerate}
    \item \textbf{Confusing AK (α=1) with Solow (α<1)}
    \begin{itemize}
        \item AK: constant MPK → growth can be sustained without tech progress
        \item Solow: diminishing MPK → need exogenous $g$ for long-run growth
    \end{itemize}

    \item \textbf{Forgetting AK Has No Transitional Dynamics}
    \begin{itemize}
        \item In AK, growth rate is constant immediately
        \item No convergence to steady state (unlike Solow)
        \item $\dot{k} = sAk - \delta k$ has no interior steady state
    \end{itemize}

    \item \textbf{Missing the Difference Between Private and Social Returns}
    \begin{itemize}
        \item With externalities $\eta > 0$: private MPK = $\alpha A$, social = $(\alpha+\eta)A$
        \item Market under-accumulates (too little saving)
    \end{itemize}

    \item \textbf{Confusing Scale Effects}
    \begin{itemize}
        \item Romer: $\gamma \propto L$ (more population → faster growth)
        \item Solow: no scale effects (growth independent of $L$)
        \item Key testable implication: $d\gamma/dL > 0$
    \end{itemize}

    \item \textbf{Forgetting Monopolistic Competition in Romer}
    \begin{itemize}
        \item Need monopoly profits to fund R\&D
        \item Without market power, no incentive to innovate
        \item Price = $1/\alpha >$ marginal cost = $\alpha$
    \end{itemize}

    \item \textbf{Mixing Up Markup Formula}
    \begin{itemize}
        \item Markup = $1/\alpha$ (not $\alpha$)
        \item Price = MC × markup = $\alpha \times (1/\alpha) = 1$
    \end{itemize}

    \item \textbf{Confusing Growth Rates in Different Models}
    \begin{itemize}
        \item AK: $\gamma = sA - \delta$ (exogenous savings) OR $\gamma = (A-\delta-\rho)/\sigma$ (Ramsey)
        \item Romer: $\gamma = (\lambda(1-\alpha)L - \rho)/\sigma$
        \item Don't mix them up!
    \end{itemize}

    \item \textbf{Jones Fix Misconception}
    \begin{itemize}
        \item Jones eliminates scale effects but needs $n > 0$ for growth
        \item Without population growth, no sustained growth
        \item This is semi-endogenous, not fully endogenous
    \end{itemize}
\end{enumerate}

\subsection*{Tips}

\begin{itemize}
    \item \textbf{Model Identification:} Always check: is $\alpha < 1$ (Solow), $\alpha = 1$ (AK), or monopolistic competition (Romer)?
    \item \textbf{Competitive Markets:} Remember: $r =$ MPK $-\delta$ in competitive markets
    \item \textbf{Romer Growth:} In Romer, growth = growth of $M$ (varieties)
    \item \textbf{Scale Effects Test:} Scale effects: $d\gamma/dL > 0$ is the key test
\end{itemize}
