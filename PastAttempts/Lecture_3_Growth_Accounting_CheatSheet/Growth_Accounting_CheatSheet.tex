\section{Notation and Definitions}
%==============================================================================

\subsection*{Core Variables}
\begin{itemize}
    \item $Y_t$: Aggregate output
    \item $K_t$: Aggregate capital stock
    \item $L_t$: Labor force
    \item $B_t$: Technology level (TFP)
    \item $A_t$: Labor productivity (often used interchangeably with $B_t$)
    \item $H_t$: Human capital-adjusted labor ($H_t = e^{\phi(E_t)}L_t$)
    \item $E_t$: Years of schooling
    \item $w_t$: Real wage
    \item $r_t$: Real interest rate
    \item $\delta$: Depreciation rate
    \item $I_t$: Investment
\end{itemize}

\subsection*{Growth Rates}
\begin{itemize}
    \item $\gammay \equiv \dot{Y}/Y$: Output growth rate
    \item $\gammaA \equiv \dot{A}/A$: TFP growth rate
    \item $\gammaK \equiv \dot{K}/K$: Capital growth rate
    \item $\gammaL \equiv \dot{L}/L$: Labor growth rate
\end{itemize}

\subsection*{Key Ratios}
\begin{itemize}
    \item $\ky$: Capital-output ratio
    \item $\alpha$: Capital share (typically $1/3$)
    \item $1-\alpha$: Labor share
\end{itemize}

%==============================================================================
\section{Growth Accounting Framework}
%==============================================================================

\subsection*{Primal Approach: Basic Setup}

Production function (CRS):
\[
Y_t = F(K_t, B_t L_t)
\]

Taking logs and time derivatives:
\begin{derivationbox}
\textbf{From production function to growth accounting:}

\[
\frac{\dot{Y}}{Y} = \frac{F_K K}{F}\frac{\dot{K}}{K} + \frac{F_L L}{F}\frac{\dot{L}}{L} + \frac{F_B B}{F}\frac{\dot{B}}{B}
\]

Define output elasticities:
\[
\alpha_t = \frac{F_K K}{F}, \quad 1-\alpha_t = \frac{F_L L}{F}
\]

Under CRS (Euler's theorem): $\alpha_t + (1-\alpha_t) = 1$

\[
\boxed{\gammay = \gammaA + \alpha_t \gammaK + (1-\alpha_t)\gammaL}
\tag{Growth-Accounting}
\]
\end{derivationbox}

\subsection*{Solow Residual}

Everything except measured capital and labor goes into TFP:
\[
\boxed{\gammaA = \gammay - \alpha_t \gammaK - (1-\alpha_t)\gammaL}
\tag{Solow-Residual}
\]

\begin{resultbox}
\textbf{Components of $\gamma_A$ (what's inside the black box):}
\begin{enumerate}
    \item True technology progress
    \item Misallocation (market efficiency, regulations, cronyism)
    \item Mismeasurement of capital and labor quality
    \item Returns to scale violations
    \item Externalities
\end{enumerate}
\end{resultbox}

%==============================================================================
\section{Measuring Capital Stock}
%==============================================================================

\subsection*{Perpetual Inventory Method}

\begin{derivationbox}
\textbf{Law of motion for capital:}

\[
K_{t+1} = (1-\delta)K_t + I_t
\tag{PIM-Motion}
\]

Where:
\begin{itemize}
    \item $I_t$: Investment (from national accounts)
    \item $\delta$: Depreciation rate (easy to estimate)
    \item $K_0$: Initial capital stock (most complicated part)
\end{itemize}

With constant growth assumption for initial period:
\[
K_0 = \frac{I_0}{g + \delta}
\tag{PIM-Initial}
\]

where $g$ is the steady-state growth rate.
\end{derivationbox}

%==============================================================================
\section{Cobb-Douglas Special Case}
%==============================================================================

With Cobb-Douglas production $Y = K^\alpha (BL)^{1-\alpha}$:

\begin{resultbox}
\begin{align*}
\gammay &= \gammaA + \alpha\gammaK + (1-\alpha)\gammaL \tag{CD-Growth-Accounting} \\
\gammaA &= \gammay - \alpha\gammaK - (1-\alpha)\gammaL
\end{align*}
\end{resultbox}

Alternative decomposition (output per capita):
\[
\frac{Y}{N} = B \frac{L}{N} \left(\frac{K}{Y}\right)^{\frac{\alpha}{1-\alpha}}
\tag{CD-Decomposition}
\]

Growth rates:
\[
\gamma_{Y/N} = \gamma_B + \gamma_{L/N} + \frac{\alpha}{1-\alpha}\gamma_{K/Y}
\tag{CD-PerCapita}
\]

\section{Dual Approach}
%==============================================================================

\subsection*{From Primal to Dual}

Starting from national accounting identity (no profits):
\[
Y_t = w_t L_t + r_t K_t
\]

Taking log derivatives:
\[
\frac{\dot{Y}}{Y} = \alpha_K\left(\frac{\dot{r}}{r} + \frac{\dot{K}}{K}\right) + \alpha_L\left(\frac{\dot{w}}{w} + \frac{\dot{L}}{L}\right)
\tag{Dual-Accounting}
\]

TFP growth via dual approach:
\[
\boxed{\gammaA = \alpha_K \frac{\dot{r}}{r} + \alpha_L \frac{\dot{w}}{w}}
\tag{Dual-TFP}
\]

\begin{resultbox}
\textbf{Comparison: Primal vs. Dual}

\begin{itemize}
    \item Primal: Uses production function elasticities ($F_K K/F$)
    \item Dual: Uses factor cost shares ($rK/Y$, $wL/Y$)
    \item Should coincide in theory, differ in practice
    \item Dual is more robust (no functional form assumptions)
\end{itemize}
\end{resultbox}

\subsection*{Dual vs Primal: Country Evidence}

\begin{center}

\midrule

\bottomrule
\end{tabular}
\end{center}

\section{Factor Shares vs. Elasticities}
%==============================================================================

\subsection*{The Key Insight}

Under perfect competition, factor shares equal output elasticities:
\[
\frac{F_x x}{Y} = \frac{w_x x}{pY}
\tag{Share-Elasticity}
\]

This is why we can use measured income shares in growth accounting.

\subsection*{Problem 1: Market Power}

With monopoly power:
\[
\frac{\epsilon_D - 1}{\epsilon_D} \frac{F_x x}{Y} = \frac{w_x x}{pY}
\tag{Market-Power}
\]

The markup is $\mu = \frac{\epsilon_D}{\epsilon_D - 1}$.

With profits, elasticity = cost share (not revenue share).

\subsection*{Problem 2: Fixed Costs}

If there are fixed costs $\Gamma$:
\[
\frac{F_x x}{Y} = \frac{w_x x}{pY} \neq \frac{w_x(x+\Gamma)}{pY}
\tag{Fixed-Costs}
\]

Factor share $\neq$ elasticity when fixed costs exist.

\subsection*{Problem 3: Externalities}

Private marginal product $\neq$ social marginal product with externalities.

\begin{resultbox}
\textbf{Required conditions for Share = Elasticity:}
\begin{enumerate}
    \item Perfect competition (no market power)
    \item Variable costs only (no fixed costs)
    \item No externalities (private = social MP)
\end{enumerate}
\end{resultbox}

\section{Development Accounting}
%==============================================================================

\subsection*{Framework}

Cross-sectional comparison at a point in time:
\[
\log Y_i = \log A_i + \alpha \log K_i + (1-\alpha)\log L_i
\tag{Dev-Accounting}
\]

Goal: Decompose income differences into:
\begin{itemize}
    \item Factor quantities ($K$, $L$)
    \item TFP ($A$)
\end{itemize}

Key result (Caselli 2005): TFP differences account for $>50\%$ of income variation.

\section{Hall-Jones Framework}
%==============================================================================

\subsection*{Production Function}

\[
Y_i = K_i^\alpha (A_i H_i)^{1-\alpha}
\tag{HJ-Production}
\]

where:
\begin{itemize}
    \item $K_i$: Physical capital
    \item $H_i = e^{\phi(E_i)}L_i$: Human capital
    \item $A_i$: Labor productivity (TFP)
    \item $E_i$: Years of schooling
\end{itemize}

\subsection*{Returns to Education}

\[
\phi(s) = 
\begin{cases}
0.134 \times s & \text{first 4 years} \\
0.134 \times 4 + 0.101 \times (s-4) & \text{years 4-8} \\
0.134 \times 4 + 0.101 \times 4 + 0.068 \times (s-8) & \text{year 9+} \\
\end{cases}
\tag{Returns-Education}
\]

Output per worker:
\[
y_i = \left(\frac{K_i}{Y_i}\right)^{\alpha/(1-\alpha)} h_i A_i
\tag{HJ-PerWorker}
\]

Measure TFP as residual:
\[
A_i = \frac{y_i}{\left(\frac{K_i}{Y_i}\right)^{\alpha/(1-\alpha)}h_i}
\tag{HJ-TFP}
\]

\section{Institutions and TFP}
%==============================================================================

Hypothesis: Better institutions $\Rightarrow$ higher TFP $\Rightarrow$ higher income.

Basic regression:
\[
\log y_i = \gamma + \beta S_i + \epsilon_i
\tag{Institutions-Reg}
\]

Problem: Reverse causality (income $\to$ institutions).

Solution: Instrument for institutions with:
\begin{itemize}
    \item Distance from equator
    \item Percentage speaking Western languages
    \item Predicted trade share (Frankel-Romer)
\end{itemize}

Results (Hall-Jones 1999):
\begin{center}
\begin{tabular}{lccc}
\toprule
Specification & Social Infrastructure & $\hat{\sigma}_z$ & OverID p-value \\
\midrule
Main specification & 5.14 & 0.84 & 0.256 \\
Instruments: Dist, FR & 5.00 & 0.82 & 0.208 \\
OLS & 3.29 & 0.70 & — \\
\bottomrule
\end{tabular}
\end{center}

\begin{resultbox}
\textbf{Conclusion:} Institutions (social infrastructure) strongly correlated with TFP and output per worker.
\end{resultbox}

\section{Misallocation: Hsieh-Klenow (2009)}
%==============================================================================

\subsection*{Setup}

Monopolistic competition with heterogeneous productivity $A_i$:
\[
y_i = A_i k_i^\alpha l_i^{1-\alpha}
\tag{HK-Production}
\]

CES preferences over varieties:
\[
C = \left(\int_i c_i^{\frac{\sigma-1}{\sigma}}\right)^{\frac{\sigma}{\sigma-1}}
\tag{CES-Preferences}
\]

Optimal pricing (monopolistic competition):
\[
p_i = \frac{\sigma}{\sigma-1} \frac{x}{A_i}
\tag{Pricing}
\]
where marginal cost $x = \left(\frac{r}{\alpha}\right)^\alpha \left(\frac{w}{1-\alpha}\right)^{1-\alpha}$.

Output relative to productivity:
\[
y_i \propto A_i^{\sigma}
\tag{HK-Output}
\]

\subsection*{Distortions}

Add firm-specific distortions:
\[
\max_{y_i,k_i,l_i} (1-\tau_i^Y)p_i y_i - wl_i - (1+\tau_i^K)rk_i
\tag{Distorted-Problem}
\]

Distorted capital-labor ratio:
\[
\frac{k_i}{l_i} = \frac{\alpha}{1-\alpha}\frac{w}{r}\frac{1}{1+\tau_i^K}
\tag{Distorted-KL}
\]

Distorted output:
\[
y_i \propto A_i^{\sigma} \left(\frac{1-\tau_i^Y}{1+\tau_i^K}\right)^{\sigma}
\tag{Distorted-Output}
\]

\subsection*{TFPR and TFPQ}

Define Total Factor Productivity Revenue:
\[
\tfpr_i \equiv p_i A_i = \frac{\sigma}{\sigma-1}x\frac{(1+\tau_i^K)^\alpha}{1-\tau_i^Y}
\tag{TFPR}
\]

Aggregate TFPQ:
\[
\tfpQ = \left[\int_j \left(A_j \frac{\tfpr}{\tfpr_j}\right)^{\sigma-1}\right]^{\frac{1}{\sigma-1}}
\tag{TFPQ}
\]

Key insight: Dispersion in $\tfpr$ reduces aggregate productivity.

\section{Common Pitfalls}
%==============================================================================

\begin{enumerate}
    \item \textbf{Confusing factor shares with elasticities:} Only equal under perfect competition, no fixed costs, no externalities.
    
    \item \textbf{Attributing causality:} Growth accounting is accounting, not causation. Capital accumulation may itself be driven by TFP.
    
    \item \textbf{Ignoring quality changes:} Labor quality (education) and capital quality (equipment quality) matter significantly.
    
    \item \textbf{Using wrong growth rates:} Remember technology growth $x$ affects effective variables.
    
    \item \textbf{Measuring capital incorrectly:} Perpetual inventory requires good initial $K_0$ and proper $\delta$.
    
    \item \textbf{Treating TFP as pure technology:} It contains misallocation, measurement error, and externalities.
    
    \item \textbf{Assuming Cobb-Douglas:} Elasticity of substitution = 1 may not hold.
    
    \item \textbf{Ignoring dual-primal discrepancies:} Different methods give different results in practice.
\end{enumerate}

\section{Quick Reference Formulas}

\begin{resultbox}
\begin{align*}
\text{Growth Accounting:} & \quad \gammay = \gammaA + \alpha\gammaK + (1-\alpha)\gammaL \\

\end{align*}
\end{resultbox}

\section{Problem-Solving Templates}
%==============================================================================

\subsection*{Template 1: Basic Growth Accounting}

\begin{enumerate}
    \item Identify production function and parameters
    \item Compute growth rates of $Y$, $K$, $L$
    \item Use factor share $\alpha$ (typically $1/3$)
    \item Calculate Solow residual: $\gammaA = \gammay - \alpha\gammaK - (1-\alpha)\gammaL$
    \item Interpret: what fraction of growth comes from TFP vs. factors
\end{enumerate}

\subsection*{Template 2: Development Accounting}

\begin{enumerate}
    \item Get data on $Y/L$, $K/Y$, $H/L$ for countries
    \item Use $\alpha = 1/3$
    \item Compute $A_i = y_i / [(K/Y)^{\alpha/(1-\alpha)} h_i]$
    \item Compare TFP across countries
    \item Decompose income gap into factors vs. TFP
\end{enumerate}

\subsection*{Template 3: Misallocation Analysis (HK)}

\begin{enumerate}
    \item Estimate TFPR dispersion across firms
    \item Compute aggregate TFPQ from firm productivities
    \item Calculate counterfactual: equalize TFPR
    \item Compute TFP gain: $(A_{counterfactual}/A_{actual}) - 1$
\end{enumerate}
